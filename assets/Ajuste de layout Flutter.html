dart
import 'package:flutter/material.dart';
import '../../domain/entities/user.dart';

class UserCard extends StatefulWidget {
  final User user;

  const UserCard({super.key, required this.user});

  @override
  // ignore: library_private_types_in_public_api
  _UserCardState createState() => _UserCardState();
}

class _UserCardState extends State<UserCard>
    with SingleTickerProviderStateMixin {
  bool _isExpanded = false;

  // Função para ofuscar o e-mail (substitui letras e números por *)
  String _obfuscateEmail(String email) {
    return email.replaceAll(RegExp(r'[a-zA-Z0-9]'), '*');
  }

  @override
  Widget build(BuildContext context) {
    // Caso o card esteja expandido, exibe o e-mail completo, senão, ofusca.
    final emailToShow =
        _isExpanded ? widget.user.email : _obfuscateEmail(widget.user.email);

    return Card(
      elevation: 4,
      margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
      shape: RoundedRectangleBorder(borderRadius: BorderRadius.circular(16)),
      child: InkWell(
        borderRadius: BorderRadius.circular(16),
        onTap: () {
          setState(() {
            _isExpanded = !_isExpanded;
          });
        },
        child: AnimatedSize(
          duration: const Duration(milliseconds: 300),
          curve: Curves.easeInOut,
          child: Padding(
            padding: const EdgeInsets.all(16),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.center,
              children: [
                // Exibe a imagem (avatar)
                Container(
                  decoration: BoxDecoration(
                    shape: BoxShape.circle,
                    border: Border.all(
                      color: Theme.of(context).primaryColorLight,
                      width: 2,
                    ),
                  ),
                  child: CircleAvatar(
                    backgroundImage: NetworkImage(widget.user.avatar),
                    radius: 30,
                  ),
                ),
                const SizedBox(height: 8),
                // Exibe o nome
                Text(
                  widget.user.name,
                  style: const TextStyle(
                    fontSize: 18,
                    fontWeight: FontWeight.bold,
                  ),
                  textAlign: TextAlign.center,
                ),
                const SizedBox(height: 12),
                // Exibe o e-mail (ofuscado ou completo conforme _isExpanded)
                Text(
                  emailToShow,
                  style: const TextStyle(
                    fontSize: 16,
                    color: Colors.grey,
                  ),
                  textAlign: TextAlign.center,
                ),
              ],
            ),
          ),
        ),
      ),
    );
  }
}

Explicação:

  • Ofuscação do e-mail:
    A função _obfuscateEmail substitui todos os caracteres alfanuméricos do e-mail por asteriscos. Assim, enquanto o card não for clicado, o e-mail aparece ofuscado.

  • Comportamento do clique:
    Ao clicar no card (InkWell), a variável _isExpanded é invertida, fazendo com que o widget seja reconstruído e o e-mail seja exibido completo ou ofuscado, conforme o estado.

  • Animação:
    O AnimatedSize garante uma transição suave na mudança de tamanho do conteúdo, caso a alteração de conteúdo afete o layout.

Dessa forma, a imagem e o nome ficam sempre visíveis, enquanto o e-mail permanece ofuscado até que o usuário toque no card para revelá-lo.

Aqui está um exemplo de como remover o AppBar e substituí-lo por um cabeçalho customizado dentro do corpo da tela:

dart
import 'package:flutter/material.dart';
import 'package:get/get.dart';
import 'package:luza_lista_usuarios/theme/app_colors.dart';
import '../viewmodels/user_viewmodel.dart';
import '../widgets/user_card.dart';

class UserListScreen extends StatelessWidget {
  const UserListScreen({super.key});

  @override
  Widget build(BuildContext context) {
    final controller = Get.put(UserViewModel(Get.find()));

    return Scaffold(
      // Removemos o appBar e utilizamos um cabeçalho customizado
      body: SafeArea(
        child: Column(
          children: [
            // Cabeçalho customizado
            Container(
              width: double.infinity,
              padding: const EdgeInsets.symmetric(vertical: 16, horizontal: 16),
              color: AppColors.primary,
              child: Column(
                crossAxisAlignment: CrossAxisAlignment.start,
                children: const [
                  Text(
                    'Lista de Usuários',
                    style: TextStyle(
                      fontSize: 20,
                      fontWeight: FontWeight.bold,
                      color: Colors.white,
                    ),
                  ),
                  SizedBox(height: 4),
                  Text(
                    'Visualize e gerencie usuários de forma simples e eficiente.',
                    style: TextStyle(
                      fontSize: 14,
                      fontWeight: FontWeight.normal,
                      color: Colors.white70,
                    ),
                  ),
                ],
              ),
            ),
            // Conteúdo principal com a lista de usuários
            Expanded(
              child: Obx(() {
                if (controller.isLoading.value) {
                  return const Center(child: CircularProgressIndicator());
                }
                return RefreshIndicator(
                  onRefresh: controller.fetchUsers,
                  child: ListView.builder(
                    padding: const EdgeInsets.only(top: 8, bottom: 8),
                    itemCount: controller.users.length,
                    itemBuilder: (context, index) {
                      final user = controller.users[index];
                      return UserCard(user: user);
                    },
                  ),
                );
              }),
            ),
          ],
        ),
      ),
    );
  }
}

  • SafeArea:
    O widget SafeArea garante que o conteúdo fique posicionado de forma adequada, evitando áreas de entalhe ou barras de status.

  • Organização com Column:
    O cabeçalho customizado é exibido no topo, seguido pelo conteúdo principal (lista de usuários) que ocupa o restante do espaço graças ao widget Expanded.

Com essa abordagem, você substitui o AppBar por um cabeçalho customizado, mantendo a funcionalidade e melhorando a flexibilidade do layout.
